
&НаСервере
Процедура ЗаполнитьПоМакетуНаСервере()
	Макет = Справочники.NSI_СправочникОКОПФ.ПолучитьМакет("ОКОПФ");
	ОбластьЗаполнения = Макет.ПолучитьОбласть("Классификатор");
	ПоследняяСтрока = ОбластьЗаполнения.ВысотаТаблицы;
	Для Ц=1 По ПоследняяСтрока Цикл
		СозданныйСправочник = Справочники.NSI_СправочникОКОПФ.НайтиПоКоду(ОбластьЗаполнения.Область(Ц,2).Текст);
		Если ЗначениеЗаполнено(СозданныйСправочник) Тогда
			СправочникОбъект = СозданныйСправочник.ПолучитьОбъект();
		Иначе
			СправочникОбъект = Справочники.NSI_СправочникОКОПФ.СоздатьЭлемент();
			СправочникОбъект.Код = ОбластьЗаполнения.Область(Ц,2).Текст;
		КонецЕсли;
		СправочникОбъект.СтарыйКод = ОбластьЗаполнения.Область(Ц,1).Текст;
		СправочникОбъект.Наименование = ОбластьЗаполнения.Область(Ц,3).Текст;
		СправочникОбъект.Сокращение = ОбластьЗаполнения.Область(Ц,4).Текст;
		СправочникОбъект.Записать();
	КонецЦикла;
КонецПроцедуры

&НаКлиенте
Процедура ЗаполнитьПоМакету(Команда)
	ЗаполнитьПоМакетуНаСервере();
КонецПроцедуры
