
Процедура ПриЗаписи(Отказ)
	Изменения = СтруктураИзменений.Получить();	
	Если ТипЗнч(Изменения)=Тип("ТаблицаЗначений") Тогда
		НаборЗаписей = РегистрыСведений.NSI_ИндексИзмененийВнешнихБаз.СоздатьНаборЗаписей();
		НаборЗаписей.Отбор.Изменение.Установить(Ссылка);
		Для Каждого Реквизит Из Изменения Цикл
			НоваяЗапись = НаборЗаписей.Добавить();	
			НоваяЗапись.Изменение = Ссылка;
			НоваяЗапись.ИмяИзмененногоРеквизита = Реквизит.Имя;
			НоваяЗапись.ИДСсылки = ИДСсылки;
			НоваяЗапись.ВнешняяБаза = ВнешняяБаза;
			НоваяЗапись.ДатаИзменения = ДатаИзменения;
			НоваяЗапись.ТабличнаяЧасть = Реквизит.ЭтоТаблица;
		КонецЦикла;
		НаборЗаписей.Записать();
	КонецЕсли;
КонецПроцедуры
